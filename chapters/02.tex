\sub{これ・それ・あれ・どれ}
    {
    this・that・that over there・which?

    \hl{これ}は\ruby{私}{わたし}の\ruby{本}{ほん}です。
    }

\sub{この・その・あの・どの + noun}
    {
    this・that・that over there・which?

    \hl{この}\ruby{本}{ほん}は\ruby{五百円}{ごひゃくえん}です。
    }
    
\sub{ここ・そこ・あそこ・どこ}
    {
    here・there・over there・where?

    レストランは\hl{どこ}ですか？

    \hl{あそこ}です。
    }

\sub{だれのnoun}
    {
    これは\hl{だれの}かばんですか。

    それはメアリーさんのかばんです。
    }
    
\sub{Nounも}
    {
    メアリーさんは学生です。

    \ruby{私}{わたし}\hl{も}\ruby{学生}{がくせい}です。
    }
    
\sub{Nounじゃないです}
    {メアリーさんは\ruby{先生}{せんせい}\hl{じゃないです}。}

\sub{〜ね・〜よ}
    {
    \begin{enumerate}
        \item そうです\hl{ね}。
        \item おすしが\ruby{大好}{だいす}きです\hl{よ}。
    \end{enumerate}
    }
